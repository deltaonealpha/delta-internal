<script>
// Pred. Preferences - Redir Functions
    
function hmsbts_unreads_rls() {
if ("meter_hmsbts_rls" in localStorage) {
var meter_hmsbts_rls = localStorage.getItem("meter_hmsbts_rls");
meter_hmsbts_rls++;
localStorage.setItem("meter_hmsbts_rls", meter_hmsbts_rls);
} else {
localStorage.setItem("meter_hmsbts_rls", 1);
}
gotomyrlpage();
}

function hmsbts_writingprompts_redir() {
if ("meter_hmsbts_writingprompts" in localStorage) {
var meter_hmsbts_writingprompts = localStorage.getItem("meter_hmsbts_writingprompts");
meter_hmsbts_writingprompts++;
localStorage.setItem("meter_hmsbts_writingprompts", meter_hmsbts_writingprompts);
} else {
localStorage.setItem("meter_hmsbts_writingprompts", 1);
}
gotowritingprompts();
}
    
// Pred. Pref - Override and Dismiss Functions
function override_rls() {
localStorage.setItem("override_pp1s_rls", "true");
localStorage.removeItem("antioverride_pp1s_rls");
location.reload();
}
    
function override_writingprompts() {
localStorage.setItem("override_pp1s_writingprompts", "true");
localStorage.removeItem("antioverride_pp1s_writingprompts");
location.reload();
}

function dismiss_rls() {
localStorage.setItem("dismiss_pp1s_rls", "true");
document.getElementById("predpref_hmsbts_rls").style.display = "none"; 
}

function dismiss_writingprompts() {
localStorage.setItem("dismiss_pp1s_writingprompts", "true");
document.getElementById("predpref_hmsbts_writingprompts").style.display = "none"; 
}
</script>
